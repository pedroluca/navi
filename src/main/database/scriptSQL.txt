CREATE TABLE Usuario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE,
    nome VARCHAR(200),
    email VARCHAR(150),
    senha VARCHAR(500),
    telefone VARCHAR(11),
    sexo CHAR(1),
    id_fase_atual INT DEFAULT(1),
    nome_do_responsavel VARCHAR(200),
    is_adm BOOLEAN NOT NULL DEFAULT(FALSE),
    xp_atual INT NOT NULL DEFAULT(0)
);

CREATE TABLE Questao (
    id INT PRIMARY KEY AUTO_INCREMENT,
    assunto ENUM('Verbo', 'Adverbio', 'Substantivo', 'Pronome', 'Adjetivo'),
    enunciado VARCHAR(1000)
);

CREATE TABLE Alternativa (
    id INT PRIMARY KEY AUTO_INCREMENT,
    texto VARCHAR(200),
    is_correta BOOLEAN DEFAULT(FALSE),
    id_questao INT
);
